018package org.fcrepo.auth.roles.basic;
019
020import java.security.Principal;
021import java.util.Set;
022
023import javax.jcr.Session;
024
025import org.fcrepo.auth.roles.common.AbstractRolesAuthorizationDelegate;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029/**
030 * @author Gregory Jansen
031 */
032@Deprecated
033public class BasicRolesAuthorizationDelegate extends AbstractRolesAuthorizationDelegate {
034
035    public static final String EVERYONE_NAME = "EVERYONE";
036
037    /**
038     * The security principal for every request, that represents the "EVERYONE" user.
039     */
040    private static final Principal EVERYONE = new Principal() {
041
042        @Override
043        public String getName() {
044            return BasicRolesAuthorizationDelegate.EVERYONE_NAME;
045        }
046
047        @Override
048        public String toString() {
049            return getName();
050        }
051
052    };
053
054    private static final Logger LOGGER = LoggerFactory
055            .getLogger(BasicRolesAuthorizationDelegate.class);
056
057    /*
058     * (non-Javadoc)
059     * @see org.fcrepo.auth.roles.AbstractRolesAuthorizationDelegate#rolesHavePermission(final
060     * String absPath, final String[] actions, final Set<String> roles)
061     */
062    @Override
063    public boolean rolesHavePermission(final Session userSession,
064            final String absPath,
065            final String[] actions, final Set<String> roles) {
066        LOGGER.warn("===========================");
067        LOGGER.warn("This authorization provider is deprecated and will be removed in a future release of Fedora: {}",
068                this.getClass());
069        LOGGER.warn("===========================");
070
071        if (roles.isEmpty()) {
072            LOGGER.debug("A caller without content roles can do nothing in the repository.");
073            return false;
074        }
075        if (roles.contains("admin")) {
076            LOGGER.debug("Granting an admin role permission to perform any action.");
077            return true;
078        }
079        if (roles.contains("writer")) {
080            if (absPath.contains(AUTHZ_DETECTION)) {
081                if (actions.length == 1 && "read".equals(actions[0])) {
082                    LOGGER.debug("Granting reader role permission to perform a read action.");
083                    return true;
084                } else {
085                    LOGGER.debug("Denying writer role permission to perform an action on an ACL node.");
086                    return false;
087                }
088            } else {
089                LOGGER.debug("Granting writer role permission to perform any action on a non-ACL node.");
090                return true;
091            }
092        }
093        if (roles.contains("reader")) {
094            if (actions.length == 1 && "read".equals(actions[0])) {
095                LOGGER.debug("Granting reader role permission to perform a read action.");
096                return true;
097            }
098            LOGGER.debug("Denying reader role permission to perform a non-read action.");
099            return false;
100        }
101        LOGGER.error("There are roles in session that aren't recognized by this authorization delegate: {}",
102                roles);
103        return false;
104    }
105
106    /**
107     * Get the principal that represents the "EVERYONE" user.
108     */
109    @Override
110    public Principal getEveryonePrincipal() {
111        return EVERYONE;
112    }
113
114}
