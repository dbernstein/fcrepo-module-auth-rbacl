018package org.fcrepo.auth.roles.common;
019
020import static java.util.Collections.singletonMap;
021
022import java.util.Map;
023
024import javax.ws.rs.core.UriInfo;
025
026import org.fcrepo.http.commons.api.rdf.UriAwareResourceModelFactory;
027import org.fcrepo.kernel.api.FedoraTypes;
028import org.fcrepo.kernel.api.RdfLexicon;
029import org.fcrepo.kernel.api.identifiers.IdentifierConverter;
030import org.fcrepo.kernel.api.models.FedoraResource;
031
032import org.apache.jena.rdf.model.Model;
033import org.apache.jena.rdf.model.ModelFactory;
034import org.apache.jena.rdf.model.Resource;
035
036/**
037 * Exposes access roles endpoint for any resource.
038 *
039 * @author Gregory Jansen
040 */
041public class AccessRolesResources implements UriAwareResourceModelFactory {
042
043    /*
044     * (non-Javadoc)
045     * @see org.fcrepo.http.commons.api.rdf.UriAwareResourceModelFactory#
046     * createModelForResource( org.fcrepo.kernel.api.FedoraResourceImpl,
047     * javax.ws.rs.core.UriInfo, org.fcrepo.kernel.api.rdf.IdentifierTranslator)
048     */
049    @Override
050    public Model createModelForResource(final FedoraResource resource,
051            final UriInfo uriInfo, final IdentifierConverter<Resource, FedoraResource> graphSubjects) {
052        final Model model = ModelFactory.createDefaultModel();
053        final Resource s = graphSubjects.reverse().convert(resource);
054
055        if (resource.hasType(FedoraTypes.FEDORA_RESOURCE)) {
056            final Map<String, String> pathMap =
057                    singletonMap("path", resource.getPath().substring(1));
058            final Resource acl = model.createResource(uriInfo.getBaseUriBuilder().path(
059                    AccessRoles.class).buildFromMap(pathMap).toASCIIString());
060            model.add(s, RdfLexicon.HAS_ACCESS_ROLES_SERVICE, acl);
061        }
062        return model;
063    }
064}
