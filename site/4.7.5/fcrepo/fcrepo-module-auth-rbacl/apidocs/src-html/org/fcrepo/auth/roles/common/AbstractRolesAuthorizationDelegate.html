018package org.fcrepo.auth.roles.common;
019
020import static java.util.stream.Collectors.toSet;
021import static org.fcrepo.kernel.modeshape.FedoraSessionImpl.getJcrSession;
022
023import java.security.Principal;
024import java.util.Arrays;
025import java.util.Collection;
026import java.util.Map;
027import java.util.Set;
028
029import javax.inject.Inject;
030import javax.jcr.Item;
031import javax.jcr.Node;
032import javax.jcr.NodeIterator;
033import javax.jcr.RepositoryException;
034import javax.jcr.Session;
035
036import org.fcrepo.auth.common.FedoraAuthorizationDelegate;
037import org.fcrepo.http.commons.session.SessionFactory;
038import org.fcrepo.kernel.api.FedoraSession;
039import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
040import org.modeshape.jcr.value.Path;
041import org.slf4j.Logger;
042import org.slf4j.LoggerFactory;
043
044/**
045 * Policy enforcement point for roles-based authentication
046 * @author Gregory Jansen
047 */
048public abstract class AbstractRolesAuthorizationDelegate implements FedoraAuthorizationDelegate {
049
050    private static final Logger LOGGER = LoggerFactory
051            .getLogger(AbstractRolesAuthorizationDelegate.class);
052
053    protected static final String AUTHZ_DETECTION = "/{" +
054            Constants.JcrName.NS_URI + "}";
055
056    private static final String[] REMOVE_ACTIONS = {"remove"};
057
058    @Inject
059    private AccessRolesProvider accessRolesProvider = null;
060
061    @Inject
062    private SessionFactory sessionFactory = null;
063
064    /**
065     * Gather effectives roles
066     *
067     * @param acl access control list
068     * @param principals effective principals
069     * @return set of effective content roles
070     */
071    public static Set<String> resolveUserRoles(final Map<String, Collection<String>> acl,
072                    final Collection<Principal> principals) {
073        return principals.stream().map(Principal::getName).filter(acl::containsKey)
074            .peek(principal -> LOGGER.debug("request principal matched role assignment: {}", principal))
075            .map(acl::get)
076            .flatMap(Collection::stream)
077            .collect(toSet());
078    }
079
080    @Override
081    public boolean hasPermission(final Session session, final Path absPath, final String[] actions) {
082        LOGGER.debug("Does user have permission for actions: {}, on path: {}", actions, absPath);
083        final boolean permission = doHasPermission(session, absPath, actions);
084
085        LOGGER.debug("Permission for actions: {}, on: {} = {}", actions, absPath, permission);
086        return permission;
087    }
088
089    private boolean doHasPermission(final Session session, final Path absPath, final String[] actions) {
090        final Set<String> roles;
091
092        final Principal userPrincipal = getUserPrincipal(session);
093        if (userPrincipal == null) {
094            return false;
095        }
096
097        final Set<Principal> allPrincipals = getPrincipals(session);
098        if (allPrincipals == null) {
099            return false;
100        }
101
102        try {
103            final FedoraSession internalSession = sessionFactory.getInternalSession();
104            final Map<String, Collection<String>> acl =
105                    accessRolesProvider.findRolesForPath(absPath,
106                            getJcrSession(internalSession));
107            roles = resolveUserRoles(acl, allPrincipals);
108            LOGGER.debug("roles for this request: {}", roles);
109        } catch (final RepositoryException e) {
110            throw new RepositoryRuntimeException("Cannot look up node information on " + absPath +
111                    " for permissions check.", e);
112        }
113
114        if (LOGGER.isDebugEnabled()) {
115            LOGGER.debug("roles: {}, actions: {}, path: {}", roles, actions, absPath);
116            if (actions.length > 1) { // have yet to see more than one
117                LOGGER.debug("FOUND MULTIPLE ACTIONS: {}", Arrays
118                        .toString(actions));
119            }
120        }
121
122        if (actions.length == 1 && "remove_child_nodes".equals(actions[0])) {
123            // in roles-based ACLs, the permission to remove children is
124            // conferred by earlier check for "remove_node" on the child node
125            // itself.
126            return true;
127        }
128
129        if (!rolesHavePermission(session, absPath.toString(), actions, roles)) {
130            return false;
131        }
132
133        if (actions.length == 1 && "remove".equals(actions[0])) {
134            // you must be able to delete all the children
135            // TODO make recursive/ACL-query-based check configurable
136            return canRemoveChildrenRecursive(session, absPath.toString(),
137                    allPrincipals, roles);
138        }
139        return true;
140    }
141
142    private static Principal getUserPrincipal(final Session session) {
143        final Object value = session.getAttribute(FEDORA_USER_PRINCIPAL);
144        if (value instanceof Principal) {
145            return (Principal) value;
146        }
147        return null;
148    }
149
150    @SuppressWarnings("unchecked")
151    private static Set<Principal> getPrincipals(final Session session) {
152        final Object value = session.getAttribute(FEDORA_ALL_PRINCIPALS);
153        if (value instanceof Set<?>) {
154            return (Set<Principal>) value;
155        }
156        return null;
157    }
158
159    /**
160     * @param userSession the user session
161     * @param parentPath the parent path
162     * @param allPrincipals all principals
163     * @param parentRoles the roles on the parent
164     * @return true if permitted
165     */
166    private boolean canRemoveChildrenRecursive(final Session userSession,
167                                               final String parentPath,
168                                               final Set<Principal> allPrincipals,
169                                               final Set<String> parentRoles) {
170        try {
171            final FedoraSession internalSession = sessionFactory.getInternalSession();
172            LOGGER.debug("Recursive child remove permission checks for: {}",
173                    parentPath);
174            final Item item = getJcrSession(internalSession).getItem(parentPath);
175            if (!item.isNode()) {
176                // this is a property and has no children...
177                return true;
178            }
179            final Node parent = (Node) item;
180            if (!parent.hasNodes()) {
181                return true;
182            }
183            final NodeIterator ni = parent.getNodes();
184            while (ni.hasNext()) {
185                final Node n = ni.nextNode();
186                // are there unique roles?
187                final Set<String> roles;
188                final Map<String, Collection<String>> acl = accessRolesProvider.getRoles(n, false);
189
190                if (acl != null) {
191                    roles = resolveUserRoles(acl, allPrincipals);
192                } else {
193                    roles = parentRoles;
194                }
195                if (rolesHavePermission(userSession, n.getPath(),
196                        REMOVE_ACTIONS,
197                        roles)) {
198
199                    if (!canRemoveChildrenRecursive(userSession, n.getPath(),
200                            allPrincipals, roles)) {
201                        return false;
202                    }
203                } else {
204                    LOGGER.info("Remove permission denied at {} with roles {}", n.getPath(), roles);
205                    return false;
206                }
207            }
208            return true;
209        } catch (final RepositoryException e) {
210            throw new RepositoryRuntimeException(
211                    "Cannot lookup child permission check information for " +
212                            parentPath, e);
213        }
214    }
215
216    /**
217     * Subclasses must override this method to determine permissions based on
218     * supplied roles.
219     * 
220     * @param userSession the user session
221     * @param absPath path to the object
222     * @param actions requested action
223     * @param roles effective roles for this request and content
224     * @return true if role has permission
225     */
226    public abstract boolean rolesHavePermission(final Session userSession, final String absPath,
227            final String[] actions, final Set<String> roles);
228
229}
