018package org.fcrepo.auth.roles.common;
019
020import static java.util.Collections.emptyMap;
021
022import java.util.Collection;
023import java.util.Map;
024import java.util.Set;
025
026import javax.jcr.Node;
027import javax.jcr.RepositoryException;
028import javax.jcr.Session;
029
030import org.modeshape.jcr.value.Path;
031
032/**
033 * Provides the effective access roles for authorization.
034 *
035 * @author Gregory Jansen
036 * @author whikloj
037 */
038public interface AccessRolesProvider {
039
040    public static final Map<String, Collection<String>> DEFAULT_ACCESS_ROLES = emptyMap();
041
042    /**
043     * Get the roles assigned to this Node. Optionally search up the tree for the effective roles.
044     *
045     * @param node the subject Node
046     * @param effective if true then search for effective roles
047     * @return a set of roles for each principal
048     */
049    public Map<String, Collection<String>> getRoles(final Node node, final boolean effective);
050
051    /**
052     * Assigns the given set of roles to each principal.
053     *
054     * @param node the Node to edit
055     * @param data the roles to assign
056     * @throws RepositoryException if repository exception occurred
057     */
058    public void postRoles(final Node node, final Map<String, Set<String>> data)
059            throws RepositoryException;
060
061    /**
062     * Deletes all roles assigned on this node and removes the mixin type.
063     *
064     * @param node the node to delete
065     * @throws RepositoryException if delete failed
066     */
067    public void deleteRoles(final Node node) throws RepositoryException;
068
069    /**
070     * Finds effective roles assigned to a path, using first real ancestor node.
071     *
072     * @param absPath the real or potential node path
073     * @param session session
074     * @return the roles assigned to each principal
075     * @throws RepositoryException if PathNotFoundException can not handle
076     */
077    public Map<String, Collection<String>> findRolesForPath(final Path absPath,
078            final Session session) throws RepositoryException;
079
080}
