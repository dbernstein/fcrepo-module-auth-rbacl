051/**
052 * Provides the effective access roles for roles based authorization.
053 *
054 * @author Gregory Jansen
055 * @author whikloj
056 */
057@Component
058public class RbAclAccessRolesProvider implements AccessRolesProvider {
059
060    private static final Logger LOGGER = getLogger(RbAclAccessRolesProvider.class);
061
062    /**
063     * {@inheritDoc}
064     */
065    @Override
066    public Map<String, Collection<String>> getRoles(final Node node, final boolean effective) {
067        try {
068            LOGGER.debug("Finding roles for: {}, effective={}", node.getPath(), effective);
069        } catch (final RepositoryException e) {
070            LOGGER.debug("Unable to get path! {}", e.getMessage());
071        }
072
073        final Map<String, Collection<String>> data = new HashMap<>();
074        try {
075
076            final Session session = node.getSession();
077            registerPrefixes(session);
078            if (node.isNodeType(rbaclAssignable.getQualified())) {
079                getAssignments(node, data);
080                return data;
081            }
082            if (effective) { // look up the tree
083                try {
084                    for (Node n = node.getParent(); n != null; n = n.getParent()) {
085                        if (n.isNodeType(rbaclAssignable.getQualified())) {
086                            if (LOGGER.isDebugEnabled()) {
087                                LOGGER.debug("effective roles are assigned at node: {}", n.getPath());
088                            }
089                            getAssignments(n, data);
090                            if (LOGGER.isDebugEnabled()) {
091                                for (final Map.Entry<String, Collection<String>> entry : data.entrySet()) {
092                                    LOGGER.debug("{} has role(s) {}", entry.getKey(), entry.getValue());
093                                }
094                            }
095                            return data;
096                        }
097                    }
098                } catch (final ItemNotFoundException e) {
099                    LOGGER.debug("Subject not found, using default access roles: {}", e.getMessage());
100                    return DEFAULT_ACCESS_ROLES;
101                }
102            }
103        } catch (final RepositoryException e) {
104            throw new RepositoryRuntimeException(e);
105        }
106        return null;
107    }
108
109    /**
110     * @param node
111     * @param data
112     * @throws RepositoryException
113     */
114    private void getAssignments(final Node node, final Map<String, Collection<String>> data)
115        throws RepositoryException {
116
117        if (node.isNodeType(rbaclAssignable.getQualified())) {
118            try {
119                final Node rbacl = node.getNode(JcrName.rbacl.getQualified());
120                LOGGER.debug("got rbacl: {}", rbacl);
121                for (final NodeIterator ni = rbacl.getNodes(); ni.hasNext();) {
122                    final Node assign = ni.nextNode();
123                    final String principalName =
124                            assign.getProperty(principal.getQualified())
125                                    .getString();
126                    if (principalName == null ||
127                            principalName.trim().length() == 0) {
128                        LOGGER.warn("found empty principal name on node {}",
129                                    node.getPath());
130                    } else {
131                        Collection<String> roles = data.get(principalName);
132                        if (roles == null) {
133                            roles = new ArrayList<>();
134                            data.put(principalName, roles);
135                        }
136                        for (final Value v : assign.getProperty(
137                                role.getQualified()).getValues()) {
138                            if (v == null || v.toString().trim().length() == 0) {
139                                LOGGER.warn("found empty role name on node {}",
140                                            node.getPath());
141                            } else {
142                                roles.add(v.toString());
143                            }
144                        }
145                    }
146                }
147            } catch (final PathNotFoundException e) {
148                LOGGER.info(
149                             "Found rbaclAssignable mixin without a corresponding node at {}",
150                             node.getPath());
151            }
152        }
153    }
154
155    /**
156     * {@inheritDoc}
157     */
158    @Override
159    public void postRoles(final Node node, final Map<String, Set<String>> data)
160            throws RepositoryException {
161        final Session session = node.getSession();
162        registerPrefixes(session);
163        if (!node.isNodeType(rbaclAssignable.getQualified())) {
164            node.addMixin(rbaclAssignable.getQualified());
165            LOGGER.debug("added rbaclAssignable type");
166        }
167
168        Node acl;
169
170        if (node.hasNode(rbacl.getQualified())) {
171            acl = node.getNode(rbacl.getQualified());
172            for (final NodeIterator ni = acl.getNodes(); ni.hasNext();) {
173                ni.nextNode().remove();
174            }
175        } else {
176            acl = node.addNode(rbacl.getQualified(), Rbacl.getQualified());
177        }
178
179        for (final Map.Entry<String, Set<String>> entry : data.entrySet()) {
180            final Node assign = acl.addNode(assignment.getQualified(), Assignment.getQualified());
181            assign.setProperty(principal.getQualified(), entry.getKey());
182            assign.setProperty(role.getQualified(), toArray(entry.getValue(), String.class));
183        }
184    }
185
186    /**
187     * {@inheritDoc}
188     */
189    @Override
190    public void deleteRoles(final Node node) throws RepositoryException {
191        final Session session = node.getSession();
192        registerPrefixes(session);
193        if (node.isNodeType(rbaclAssignable.getQualified())) {
194            // remove rbacl child
195            try {
196                final Node rbacl = node.getNode(JcrName.rbacl.getQualified());
197                rbacl.remove();
198            } catch (final PathNotFoundException e) {
199                LOGGER.debug("Cannot find node: {}", node, e);
200            }
201            // remove mixin
202            node.removeMixin(rbaclAssignable.getQualified());
203        }
204    }
205
206    /**
207     * {@inheritDoc}
208     */
209    @Override
210    public Map<String, Collection<String>> findRolesForPath(final Path absPath,
211            final Session session) throws RepositoryException {
212        Node node = null;
213        for (Path p = absPath; p != null; p = p.getParent()) {
214            try {
215                if (p.isRoot()) {
216                    node = session.getRootNode();
217                } else {
218                    node = session.getNode(p.getString());
219                }
220                break;
221            } catch (final PathNotFoundException e) {
222                LOGGER.trace("Cannot find node: {}, trying parent.", p, e);
223            }
224        }
225        return this.getRoles(node, true);
226    }
