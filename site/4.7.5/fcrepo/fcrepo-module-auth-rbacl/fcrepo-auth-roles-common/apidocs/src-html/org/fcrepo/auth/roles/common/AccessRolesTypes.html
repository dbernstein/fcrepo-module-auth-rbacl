018package org.fcrepo.auth.roles.common;
019
020import java.io.IOException;
021import java.net.URL;
022
023import javax.annotation.PostConstruct;
024import javax.inject.Inject;
025import javax.jcr.RepositoryException;
026import javax.jcr.nodetype.NodeType;
027import javax.jcr.nodetype.NodeTypeIterator;
028
029import org.fcrepo.http.commons.session.SessionFactory;
030import org.fcrepo.kernel.api.FedoraSession;
031import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
032import org.modeshape.jcr.api.nodetype.NodeTypeManager;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035import org.springframework.stereotype.Component;
036
037import static org.fcrepo.kernel.modeshape.FedoraSessionImpl.getJcrSession;
038
039/**
040 * @author Gregory Jansen
041 *
042 */
043@Component
044public class AccessRolesTypes {
045
046    private static final Logger LOGGER = LoggerFactory
047            .getLogger(AccessRolesTypes.class);
048
049    @Inject
050    private final SessionFactory sessionFactory = null;
051
052    /**
053     * Initialize, register role assignment node types.
054     *
055     * @throws IOException if io exception occurred
056     */
057    @PostConstruct
058    public void setUpRepositoryConfiguration() throws IOException {
059        registerNodeTypes(sessionFactory);
060    }
061
062    private void registerNodeTypes(final SessionFactory sessions) throws IOException {
063        FedoraSession session = null;
064        try {
065            session = sessions.getInternalSession();
066            final NodeTypeManager mgr =
067                    (NodeTypeManager) getJcrSession(session).getWorkspace()
068                            .getNodeTypeManager();
069            final URL cnd =
070                    AccessRoles.class
071                            .getResource("/cnd/access-control.cnd");
072            final NodeTypeIterator nti =
073                    mgr.registerNodeTypes(cnd, true);
074            while (nti.hasNext()) {
075                final NodeType nt = nti.nextNodeType();
076                LOGGER.debug("registered node type: {}", nt.getName());
077            }
078            session.commit();
079            LOGGER.debug("Registered access role node types");
080        } catch (final RepositoryException e) {
081            throw new RepositoryRuntimeException(e);
082        } finally {
083            if (session != null) {
084                session.expire();
085            }
086        }
087    }
088}
